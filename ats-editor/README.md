## Install

```bash
npm install --save ats-editor
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'ats-editor'
import 'ats-editor/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
